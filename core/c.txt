async def _build_event_vn(self, vn: VNDBVnResponse, chas: list[VNDBCharacterResponse]) -> UnrenderedData:
    items: list[RenderedItem] = []
    imgs = []
    for item in chas:
        imgs.append(
            File.buffer2base64(await self.downloader.download_once(item.image.url))
            if item.image
            else self.err
        )
        builder = Splicer.from_vndb_character()
        text = (
            builder
            .vndb_id(item.id)
            .alias(item.aliases)
            .birthday(item.birthday)
            .vns(item.vns)
            .do()
        )

        items.append(RenderedItem(
            sub_title=item.original or item.name,
            image="",
            text="<br>".join(text),
        ))
    done = await asyncio.gather(*imgs)
    for i, j in zip(items, done):
        i.image = j

    chas = RenderedBlock(
        column_info="jue",
        vns=items
    )
    main_image = await File.buffer2base64(await self.downloader.download_once(vn.image.url) if vn.image else self.err)
    main_desc = (
        Splicer.from_vndb_vn()
        .vndb_id(vn.id)
        .average(vn.average)
        .rating(vn.rating)
        .release(vn.released)
        .length(vn.length)
        .platform(vn.platform)
        .alias(vn.aliases)
        .producer(vn.developers)
        .titles(vn.titles)
        .do()
    )

    return UnrenderedData(
        title="",
        items=[chas],
        bg_image=self.bg,
        font=self.font,
        main_image=main_image,
        main_desc="<br>".join(main_desc)
    )


async def _build_event_cha(self, cha: VNDBCharacterResponse, vns: list[VNDBVnResponse]):
    pass

async def _handle_gal_event(self, response, **kwargs):
    for_vn: bool = kwargs["vn"]
    res: UnrenderedData
    if for_vn:
        resp: tuple[VNDBVnResponse, list[VNDBCharacterResponse]] = response
        res = await self._build_event_vn(resp[0], resp[1])
    else:
        resp: tuple[VNDBCharacterResponse, list[VNDBVnResponse]] = response
        res = await self._build_event_cha(resp[0], resp[1])

    res.title = "<br>".join(kwargs.get("title", "标题出错"))
    return res

self._handlers: dict[CommandType, Callable] = {
            CommandType.VN: self._handle_vn,
            CommandType.CHARACTER: self._handle_character,
            CommandType.PRODUCER: self._handle_producer,
            CommandType.EVENT: self._handle_event,
            CommandType.RANDOM: self._handle_random,
            CommandType.DOWNLOAD: self._handle_download,
            CommandType.SELECT: self._handle_select,
            CommandType.FIND: self._handle_find,
            CommandType.RECOMMEND: self._handle_random,
            CommandType.PUZZLE: self._handle_puzzle,
            CommandType.GAL_EVENT: self._handle_gal_event
        }
